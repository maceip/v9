### Step 2: readline Module
**Priority:** HIGH — Claude Code and Gemini use it for interactive prompts
**File:** New: `napi-bridge/readline.js`

```js
createInterface({ input, output, terminal, prompt }) → Interface
```

**Interface** (extends EventEmitter):
- `question(query, callback)` — write query to output, read line from input, call callback
- `prompt()` — write prompt string to output
- `close()` — emit 'close', stop reading
- `on('line', fn)` — fires for each line of input
- `on('close', fn)` — fires on close
- `write(data)` — inject data into input
- `setPrompt(prompt)` — change prompt string

Built on our existing `process.stdin` Readable + `process.stdout` Writable. Split input on `\n` to emit 'line' events.

### Step 5: async_hooks Module
**Priority:** HIGH — Claude Code uses AsyncLocalStorage for request context
**File:** New: `napi-bridge/async-hooks.js`

```js
export class AsyncLocalStorage {
  constructor() { this._store = new Map(); this._id = 0; }

  run(store, callback, ...args) {
    const id = ++this._id;
    this._store.set(id, store);
    try {
      return callback(...args);
    } finally {
      this._store.delete(id);
    }
  }

  getStore() {
    // Return the most recently set store
    // In single-threaded browser, this is always the current run() context
    const entries = [...this._store.entries()];
    return entries.length > 0 ? entries[entries.length - 1][1] : undefined;
  }

  disable() { this._store.clear(); }
}

export class AsyncResource {
  constructor(type, opts) { this.type = type; }
  runInAsyncScope(fn, thisArg, ...args) { return fn.apply(thisArg, args); }
  emitDestroy() { return this; }
}

export function executionAsyncId() { return 0; }
export function triggerAsyncId() { return 0; }
export function createHook() { return { enable() {}, disable() {} }; }
```

This is a simplified but functional implementation. `AsyncLocalStorage.run()` provides real context isolation for nested calls. The browser is single-threaded so we don't need the full async hook machinery.

### Step 7: timers/promises
**Priority:** MEDIUM — Claude Code and Gemini use it
**File:** New: `napi-bridge/timers-promises.js`

```js
export function setTimeout(delay, value, options) {
  return new Promise((resolve, reject) => {
    const id = globalThis.setTimeout(() => resolve(value), delay);
    if (options?.signal) {
      options.signal.addEventListener('abort', () => {
        globalThis.clearTimeout(id);
        reject(options.signal.reason || new DOMException('AbortError'));
      });
    }
  });
}

export function setInterval(delay, value, options) {
  // Returns an async iterable
  return {
    async *[Symbol.asyncIterator]() {
      while (true) {
        await new Promise(r => globalThis.setTimeout(r, delay));
        if (options?.signal?.aborted) return;
        yield value;
      }
    },
  };
}

export function setImmediate(value) {
  return new Promise(resolve => {
    globalThis.setTimeout(() => resolve(value), 0);
  });
}
```

### Step 8: stream/consumers
**Priority:** MEDIUM — Claude Code uses it
**File:** Extend `napi-bridge/streams.js`

```js
export async function text(stream) {
  const chunks = [];
  for await (const chunk of stream) {
    chunks.push(typeof chunk === 'string' ? chunk : new TextDecoder().decode(chunk));
  }
  return chunks.join('');
}

export async function arrayBuffer(stream) {
  const chunks = [];
  let totalLength = 0;
  for await (const chunk of stream) {
    const buf = typeof chunk === 'string' ? new TextEncoder().encode(chunk) : chunk;
    chunks.push(buf);
    totalLength += buf.length;
  }
  const result = new Uint8Array(totalLength);
  let offset = 0;
  for (const chunk of chunks) {
    result.set(chunk, offset);
    offset += chunk.length;
  }
  return result.buffer;
}

export async function json(stream) {
  const str = await text(stream);
  return JSON.parse(str);
}
```

### Step 9: worker_threads Module
**Priority:** MEDIUM — Claude Code uses it for background processing
**File:** New: `napi-bridge/worker-threads.js`

Bridge `worker_threads.Worker` to browser `Web Worker`:

```js
export class Worker extends EventEmitter {
  constructor(filename, options) {
    super();
    // Read the script from MEMFS
    const source = fs.readFileSync(filename, 'utf8');
    const blob = new Blob([source], { type: 'application/javascript' });
    this._worker = new globalThis.Worker(URL.createObjectURL(blob), { type: 'module' });

    this._worker.onmessage = (e) => this.emit('message', e.data);
    this._worker.onerror = (e) => this.emit('error', e.error || e);
    this._worker.onmessageerror = (e) => this.emit('messageerror', e);
  }

  postMessage(value, transferList) {
    this._worker.postMessage(value, transferList);
  }

  terminate() {
    this._worker.terminate();
    return Promise.resolve(0);
  }
}

export const isMainThread = true;
export const parentPort = null;
export const workerData = undefined;
export const threadId = 0;
```

Note: This won't work for scripts that use `require()` or access MEMFS from the worker. Those would need a more sophisticated bridge. For Claude Code's usage (background tasks), this basic bridge may suffice — or it may need to be stubbed if the worker depends on too much shared state.
